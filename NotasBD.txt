
-- CREAR TABLA CONTRATOS
CREATE TABLE `contratos` (
  `id` BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `user_id` BIGINT UNSIGNED NOT NULL,

  `fecha_ingreso` DATE NOT NULL,
  `fecha_inicio_contrato` DATE NOT NULL,
  `fecha_vencimiento` DATE DEFAULT NULL,  

  `cargo` VARCHAR(150) NOT NULL,
  `tipo_contrato` VARCHAR(100) DEFAULT 'Plazo Fijo',
  
  `sueldo_bruto` DECIMAL(10, 2) NOT NULL DEFAULT '0.00',
  `sueldo_neto` DECIMAL(10, 2) NOT NULL DEFAULT '0.00',
  
  `status` ENUM('Activo', 'Vencido', 'Finalizado') DEFAULT 'Activo',
  `contrato_path` VARCHAR(255) DEFAULT NULL,
  `created_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
 
  CONSTRAINT `fk_contratos_user` FOREIGN KEY (`user_id`) 
    REFERENCES `users` (`id`) ON DELETE CASCADE
);

-- 1. Tabla de saldos de vacaciones
CREATE TABLE `vacaciones` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `idContrato` BIGINT UNSIGNED NOT NULL,
  `dias_ganados` INT NOT NULL DEFAULT 0,
  `dias_tomados` INT NOT NULL DEFAULT 0,
  `dias_restantes` INT NOT NULL DEFAULT 0,
  `created_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `vacaciones_idcontrato_foreign` (`idContrato`)
);

-- 2. Tabla de detalle de vacaciones asignadas/tomadas
CREATE TABLE `vacacion_asignada` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `idVacacion` BIGINT UNSIGNED NOT NULL,
  `tipo` VARCHAR(255) DEFAULT NULL,
  `razon` TEXT DEFAULT NULL,
  `d_tomados` INT DEFAULT NULL,
  `f_inicio` DATE DEFAULT NULL,
  `observacion` TEXT DEFAULT NULL,
  `especial` TINYINT(1) DEFAULT 0, -- 1 para sí, 0 para no
  `created_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `vacacion_asignada_idvacacion_foreign` (`idVacacion`)
);





-- 1. Tabla de Tipos de Documentos
-- Permite clasificar qué se está subiendo (DNI, Antecedentes, etc.)
CREATE TABLE tipo_documentos (

-- 2. Tabla de Documentos de Usuarios (Legajo Digital)
-- Almacena la información de los archivos físicos
CREATE TABLE documentos_usuarios (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL,
    tipo_documento_id BIGINT UNSIGNED NOT NULL,
    nombre VARCHAR(255),
    ruta VARCHAR(255) NOT NULL,
    extension VARCHAR(10),
    fecha_emision DATE NULL,

    updated_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    CONSTRAINT fk_documentos_user FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    CONSTRAINT fk_documentos_tipo FOREIGN KEY (tipo_documento_id) REFERENCES tipo_documentos(id)

-- Tabla Principal de Planillas
CREATE TABLE planillas (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    contrato_id BIGINT UNSIGNED NOT NULL,
    periodo DATE NOT NULL,

    sueldo_base DECIMAL(10, 2) NOT NULL DEFAULT 0.00,

    asignacion_familiar DECIMAL(10, 2) DEFAULT 0.00,
    horas_extras DECIMAL(10, 2) DEFAULT 0.00,
    movilidad DECIMAL(10, 2) DEFAULT 0.00,
    otros_ingresos DECIMAL(10, 2) DEFAULT 0.00,
    
    otros_descuentos DECIMAL(10, 2) DEFAULT 0.00,
    
    total_pagado DECIMAL(10, 2) GENERATED ALWAYS AS (
        sueldo_base + asignacion_familiar + horas_extras + movilidad + otros_ingresos - otros_descuentos
    ) VIRTUAL,

    planilla VARCHAR(255) NULL,
    estado_pago TINYINT(1) DEFAULT 0 COMMENT '0 = Pendiente, 1 = Pagado',
    fecha_pago DATE NULL,
    numero_cuenta VARCHAR(50) NULL,
    observacion TEXT NULL,

    created_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (contrato_id) REFERENCES contratos(id) ON DELETE CASCADE
);

-- Tabla de Archivos/Documentos de Planilla
CREATE TABLE planilla_archivos (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    planilla_id BIGINT UNSIGNED NOT NULL,
    tipo ENUM('boleta', 'comprobante', 'otros') DEFAULT 'boleta',
    nombre VARCHAR(255) NOT NULL,
    ruta VARCHAR(255) NOT NULL,
    extension VARCHAR(10) NOT NULL,
    created_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (planilla_id) REFERENCES planillas(id) ON DELETE CASCADE
);

CREATE TABLE cierres_diarios (
    fecha DATE NOT NULL,
    efectivo_esperado DECIMAL(10,2) DEFAULT 0.00,
    efectivo_real DECIMAL(10,2) DEFAULT 0.00,
    pos_esperado DECIMAL(10,2) DEFAULT 0.00,
    pos_real DECIMAL(10,2) DEFAULT 0.00,
    diferencia DECIMAL(10,2) GENERATED ALWAYS AS (efectivo_real + pos_real - (efectivo_esperado + pos_esperado)) STORED,
    observacion TEXT NULL,
    auditado_por BIGINT NULL,
    estado ENUM('pendiente', 'cuadrado', 'observado') DEFAULT 'pendiente',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (fecha),
    INDEX idx_estado_fecha (estado, fecha)
);
